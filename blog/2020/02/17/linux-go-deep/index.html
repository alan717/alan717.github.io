系统的syscall在unistd.h 文件中 不通过libc的write等系统call封装可以使用

  • _syscall3
  • _syscall2
  • _syscall1

这三种函数进行调用实现

例如 

int read(int fd,char* buf,int n)
{
    long __res;
    __asm__ volatile(
    "int $0x80"
    : "=a" (__res)
    : "0" (__NR_read),"b"((long)(fd)),"c"((long)(buf)),"d"((long)(n)));

    if(__res>=0)
         return int __res;
    errno=-__res;
    return -1;
}

同时在系统的宏展开就是这样的实现: 

#define _syscall1(type,name,atype,a) \
type name(atype a) \
{ \
long __res; \
__asm__ volatile ("int $0x80" \
        : "=a" (__res) \
        : "0" (__NR_##name),"b" ((long)(a))); \
if (__res >= 0) \
        return (type) __res; \
errno = -__res; \
return -1; \
}
